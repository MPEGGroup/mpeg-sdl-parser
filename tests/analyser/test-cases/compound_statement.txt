# Simple

class A {{utf8string foo;utf8string bar;}}

==>

[global]
  A CLASS
  [A]
    [COMPOUND]
      foo VARIABLE
      bar VARIABLE


# Nested compound statements

class A {{utf8string foo;{ utf8string bar1;utf8string bar2;}}}

==>

[global]
  A CLASS
  [A]
    [COMPOUND]
      foo VARIABLE
      [COMPOUND]
        bar1 VARIABLE
        bar2 VARIABLE
