# String definition

class A {utf8string foo;}

==>

[GLOBAL]:
  A CLASS
  [CLASS] A:
    members:
      foo VARIABLE
    foo VARIABLE


# utf8list with literal

class A {utf8list foo=u"hello world";}

==>

[GLOBAL]:
  A CLASS
  [CLASS] A:
    members:
      foo VARIABLE UTF8_LIST
    foo VARIABLE UTF8_LIST


# base64 string literal

class A {base64string foo="aGVsbG8K";}

==>

[GLOBAL]:
  A CLASS
  [CLASS] A:
    members:
      foo VARIABLE BASE64
    foo VARIABLE BASE64
